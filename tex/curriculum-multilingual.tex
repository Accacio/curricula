    \ifnum#2 > \month   %anniv month is later in year
            \advance\@tempnuma by -1\relax
    \else
            \ifnum#2 = \month   %anniv month is this month
                \ifnum#1 > \day %anniv day is later in month
                    \advance\@tempnuma by -1\relax
                \fi\relax
            \fi\relax
    \fi\relax
